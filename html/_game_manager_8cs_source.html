131 private void LoadPatient(string objPath)
132 {
133 string error = string.Empty;
134
135 if (!File.Exists(objPath))
136 {
137 error = "File doesn't exist.";
138 Debug.Log("Text: " + objPath);
139 }
140
141 else
142 {
143 if (loadedPatient != null)
144 {
145 loadedPatient.SetActive(false);
146 loadedPatient = null;
147 }
148
149 loadedPatient = new OBJLoader().Load(objPath);
150
151 loadedPatient.transform.SetParent(GameObject.Find("/OperationRoom/PatientZeroSpawn").transform);
152 loadedPatient.transform.localScale = new Vector3(
153 PATIENT_BLENDER_2_UNITY_DOWNSCALE,
154 PATIENT_BLENDER_2_UNITY_DOWNSCALE,
155 PATIENT_BLENDER_2_UNITY_DOWNSCALE
156 );
157 loadedPatient.transform.localPosition = Vector3.zero;
158 loadedPatient.transform.localEulerAngles = Vector3.zero;
159 }
160
161 if (!string.IsNullOrWhiteSpace(error))
162 {
163 DisplayMessage(error);
164 }
165 }

181 private void CreateProjectCaseButtons()
182 {
183 FileInfo[] Files;
184 try
185 {
186 DirectoryInfo d = new DirectoryInfo(Environment.GetFolderPath(Environment.SpecialFolder.UserProfile) + "/Documents/VR-OMFS");
187 Files = d.GetFiles("*.json");
188
189 if (Files.Length == 0)
190 {
191 GetComponent<CaseController>().appendCaseInfoMessage(
192 "No case found in " +
193 Environment.GetFolderPath(Environment.SpecialFolder.UserProfile) +
194 "/Documents/VR-OMFS!\nMake sure the cases are in the json Format!");
195 return;
196 }
197 } catch (Exception)
198 {
199 GetComponent<CaseController>().appendCaseInfoMessage(
200 "Folder not found!\nPut your Cases into your home Folder ->\n" +
201 Environment.GetFolderPath(Environment.SpecialFolder.UserProfile) +
202 "/Documents/VR-OMFS");
203 return;
204 }
205
206 GameObject caseInfoTemplate = GameObject.Find("/Player/SteamVRObjects/LeftHand/VRUI/Elements/VRUI-ProjectCases/Container/List/Template");
207 GameObject list = GameObject.Find("/Player/SteamVRObjects/LeftHand/VRUI/Elements/VRUI-ProjectCases/Container/List");
208
209 for (int i = 0; i < Files.Length; i++)
210 {
211 GameObject copy = Instantiate(caseInfoTemplate, Vector3.zero, Quaternion.identity);
212 copy.name = Files[i].Name.Replace(".json", "");
213 copy.GetComponentInChildren<Text>().text = Files[i].Name.Replace(".json", "");
214 copy.GetComponentInChildren<LoadProjectCase>().filePath = Files[i].FullName;
215 copy.transform.SetParent(list.transform, false);
216 copy.SetActive(true);

233 private void CreatePatientModelHierarchy(ProjectCase projectCase)
234 {
235 List<GameObject> stepList = new List<GameObject>();
236
237 foreach (Transform segment in loadedPatient.transform)
238 {
239 segment.name = segment.name.Replace(TOKEN_BOT, "").Replace(TOKEN_TOP, "");
240 segment.name = segment.name.Split('_')[0];
241
242 }
243 stepList.AddRange(from Transform segment in loadedPatient.transform
244 where segment.name.Contains(TOKEN_STEP)
245 select segment.gameObject);
246 GetComponent<CaseController>().CaseLoaded(projectCase, stepList, loadedPatient, _jsonLocation);
247 }

252 public void CreateUIElements()
253 {
254 foreach (Transform listEntry in list.transform)
255 {
256 if (listEntry.name.ToLower().Contains("template")) continue;
257 if (listEntry.name.ToLower().Contains("collider")) continue;
258
259 listEntry.gameObject.SetActive(false);
260 Destroy(listEntry);
261 }
262
263 foreach (Transform segment in loadedPatient.transform)
264 {
265 if (segment.name.ToLower().Contains("template")) continue;
266 if (segment.name.ToLower().Contains("collider")) continue;
267 if (segment.name.ToLower().Contains(TOKEN_STEP.ToLower())) continue;
268
269 GameObject copy = Instantiate(segmentsTemplate, Vector3.zero, Quaternion.identity);
270 copy.name = segment.name;
271 copy.GetComponentInChildren<Text>().text = segment.name;
272 copy.GetComponentInChildren<SegmentToggle>().target = segment.gameObject;
273 copy.GetComponentInChildren<VRUIToggle>().isOn = segment.gameObject.activeInHierarchy;
274
275 copy.transform.SetParent(list.transform, false);
276 copy.SetActive(true);
277
278 if (segment.GetComponent<Renderer>().material.GetFloat("_Mode") != 3) continue;
279
280 GameObject sliderCopy = Instantiate(sliderTemplate, Vector3.zero, Quaternion.identity);
281 sliderCopy.name = segment.name;
282 sliderCopy.GetComponentInChildren<SegmentTransparencySlide>().target = segment.gameObject;
283 sliderCopy.GetComponentInChildren<SegmentTransparencySlide>().Initialize();
284 sliderCopy.transform.SetParent(list.transform, false);
285 sliderCopy.SetActive(true);
286 }

324 private void MakeSliceableSegments()
325 {
326 foreach (Transform segment in loadedPatient.transform)
327 {
328 if (segment.name.ToLower().Contains("collider")) continue;
329
330 segment.gameObject.layer = LayerMask.NameToLayer("Sliceable");
331
332 var collider = segment.gameObject.GetOrAddComponent<MeshCollider>();
333 collider.convex = true;
334
335 var sliceable = segment.gameObject.GetOrAddComponent<Sliceable>();
336 sliceable.MaterialDensity = 100;
337 sliceable.CutMaterial = cuttingMaterial;
338
339 var rigidbody = segment.gameObject.GetOrAddComponent<Rigidbody>();
340 rigidbody.isKinematic = true;
341 }
