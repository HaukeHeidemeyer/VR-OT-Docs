92 private void FixChunk(GameObject chunk, Sliceable victim, Vector3 offsetDirection)

102
103 var meshFilter = chunk.GetOrAddComponent<MeshFilter>();
104 var collider = chunk.AddComponent<MeshCollider>();
105 collider.convex = true;
106
107 sliceable.CutMaterial = victim.CutMaterial;
108 sliceable.MaterialDensity = victim.MaterialDensity;
109 rigidbody.mass = meshFilter.mesh.Volume() * sliceable.MaterialDensity;
110 rigidbody.MovePosition(chunk.transform.position + offsetDirection * afterSliceOffset);
